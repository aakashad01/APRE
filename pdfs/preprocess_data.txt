import os
import pandas as pd
import numpy as np
from sklearn.preprocessing import LabelEncoder, MinMaxScaler
import glob
import pickle

# Configuration
DATASET_DIR = r"Datasets\BCCC-CIC-IDS2017\BCCC-CIC-IDS-2017"
OUTPUT_FILE = "processed_data.csv"
SCALER_FILE = "scaler.pkl"
ENCODER_FILE = "label_encoder.pkl"
SAMPLE_SIZE = 100000  # Number of samples to take from each file to avoid OOM

def load_data(dataset_dir):
    """Loads data from CSV files in the directory."""
    all_files = glob.glob(os.path.join(dataset_dir, "*.csv"))
    df_list = []
    
    print(f"Found {len(all_files)} CSV files.")
    count=0
    for filename in all_files:
        print(f"Loading {filename}...")
        try:
            # Read a subset of data to speed up processing for demonstration/initial training
            # Remove 'nrows=SAMPLE_SIZE' to process the full dataset (warning: high memory usage)
            df = pd.read_csv(filename, nrows=SAMPLE_SIZE) 
            df_list.append(df)
            count+=1
        except Exception as e:
            print(f"Error loading {filename}: {e}")
    print(count)

    if not df_list:
        raise ValueError("No data loaded. Check dataset directory.")

    combined_df = pd.concat(df_list, ignore_index=True)
    print(f"Combined dataset shape: {combined_df.shape}")
    return combined_df

def clean_data(df):
    """Handles NaNs and Infinite values."""
    print("Cleaning data...")
    
    # Replace Infinity with NaN
    df.replace([np.inf, -np.inf], np.nan, inplace=True)
    
    # Check for NaNs
    nan_count = df.isnull().sum().sum()
    print(f"NaNs found: {nan_count}")
    
    if nan_count > 0:
        # Drop rows with NaNs
        df.dropna(inplace=True)
        print("Dropped rows with NaNs.")
        
    # Remove duplicates
    initial_rows = df.shape[0]
    df.drop_duplicates(inplace=True)
    print(f"Removed {initial_rows - df.shape[0]} duplicate rows.")
    
    return df

def preprocess(df):
    """Encodes labels and scales features."""
    print("Preprocessing...")
    
    # Identify label column (assuming 'label' is the column name based on previous analysis)
    # Adjust if the column name is different (e.g., ' Label')
    label_col = 'label' 
    
    # Strip whitespace from column names just in case
    df.columns = df.columns.str.strip()
    
    if label_col not in df.columns:
        raise ValueError(f"Label column '{label_col}' not found in dataset.")

    # Separate features and target
    X = df.drop(columns=[label_col])
    y = df[label_col]
    
    # Drop non-numeric columns from features (like IPs, timestamps if present)
    # Based on previous view_file, 'flow_id', 'timestamp', 'src_ip', 'dst_ip' are strings
    cols_to_drop = ['flow_id', 'timestamp', 'src_ip', 'dst_ip']
    existing_cols_to_drop = [col for col in cols_to_drop if col in X.columns]
    if existing_cols_to_drop:
        print(f"Dropping non-numeric columns: {existing_cols_to_drop}")
        X.drop(columns=existing_cols_to_drop, inplace=True)

    # Label Encoding
    print("Encoding labels...")
    le = LabelEncoder()
    y_encoded = le.fit_transform(y)
    
    # Save encoder
    with open(ENCODER_FILE, 'wb') as f:
        pickle.dump(le, f)
    print(f"Label encoder saved to {ENCODER_FILE}")
    print(f"Classes: {le.classes_}")

    # Feature Scaling
    print("Scaling features...")
    # Ensure all remaining columns are numeric
    X = X.select_dtypes(include=[np.number])
    
    scaler = MinMaxScaler(feature_range=(0, 1))
    X_scaled = scaler.fit_transform(X)
    
    # Save scaler
    with open(SCALER_FILE, 'wb') as f:
        pickle.dump(scaler, f)
    print(f"Scaler saved to {SCALER_FILE}")

    # Reconstruct DataFrame
    processed_df = pd.DataFrame(X_scaled, columns=X.columns)
    processed_df['label'] = y_encoded
    
    return processed_df

def main():
    try:
        df = load_data(DATASET_DIR)
        df = clean_data(df)
        processed_df = preprocess(df)
        
        print(f"Saving processed data to {OUTPUT_FILE}...")
        processed_df.to_csv(OUTPUT_FILE, index=False)
        print("Done!")
        
    except Exception as e:
        print(f"An error occurred: {e}")

if __name__ == "__main__":
    main()
